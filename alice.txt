SELECT 
    MGR.SALESREP_NAME AS MANAGER_NAME,
    MGR.SALARY AS MANAGER_SALARY
FROM 
    SALESREP MGR
WHERE 
    MGR.SALARY > (
        SELECT 
            AVG(EMP.SALARY)
        FROM 
            SALESREP EMP
        WHERE 
            EMP.MGR = MGR.SALESREP_ID
    );
